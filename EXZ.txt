@Composable
fun VolumeSliderCard() {
    // Состояние громкости
    var volumeValue by remember { mutableFloatStateOf(65f) }
    
    // Определяем цвет в зависимости от значения
    val textColor = when {
        volumeValue == 0f -> Color(0xFFF44747) // Красный
        volumeValue in 1f..50f -> Color(0xFFFFC107) // Желтый
        volumeValue in 51f..100f -> Color(0xFF4CAF50) // Зеленый
        else -> Color(0xFFFFC107)
    }
    
    // Определяем иконку
    val volumeIcon: ImageVector = when {
        volumeValue == 0f -> Icons.Filled.VolumeOff
        volumeValue in 1f..50f -> Icons.Filled.VolumeDown
        volumeValue in 51f..100f -> Icons.Filled.VolumeUp
        else -> Icons.Filled.VolumeUp
    }
    
    Card(
        modifier = Modifier.fillMaxWidth(),
        shape = RoundedCornerShape(20.dp),
        colors = CardDefaults.cardColors(
            containerColor = Color(0xFF2B2B3C)
        ),
        elevation = CardDefaults.cardElevation(
            defaultElevation = 6.dp
        )
    ) {
        Column(
            modifier = Modifier
                .fillMaxWidth()
                .padding(20.dp),
            horizontalAlignment = Alignment.CenterHorizontally,
            verticalArrangement = Arrangement.spacedBy(16.dp)
        ) {
            // Иконка динамика
            Icon(
                imageVector = volumeIcon,
                contentDescription = "Уровень громкости",
                modifier = Modifier.size(52.dp),
                tint = textColor
            )
            
            // Слайдер
            Slider(
                value = volumeValue,
                onValueChange = { volumeValue = it },
                valueRange = 0f..100f,
                steps = 99,
                modifier = Modifier.fillMaxWidth(),
                colors = SliderDefaults.colors(
                    thumbColor = Color.White,
                    activeTrackColor = textColor,
                    inactiveTrackColor = Color(0xFF3C3C4C),
                    activeTickColor = Color.Transparent,
                    inactiveTickColor = Color.Transparent
                )
            )
            
            // Метки диапазона
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.SpaceBetween
            ) {
                Text("0", color = Color.White.copy(alpha = 0.4f), fontSize = 10.sp)
                Text("50", color = Color.White.copy(alpha = 0.4f), fontSize = 10.sp)
                Text("100", color = Color.White.copy(alpha = 0.4f), fontSize = 10.sp)
            }
            
            // Значение
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.SpaceBetween,
                verticalAlignment = Alignment.Baseline
            ) {
                Text(
                    text = "Media Volume",
                    color = Color.White.copy(alpha = 0.5f),
                    fontSize = 11.sp,
                    fontWeight = FontWeight.Medium,
                    letterSpacing = 0.5.sp
                )
                Text(
                    text = "${volumeValue.toInt()}",
                    color = textColor,
                    fontSize = 28.sp,
                    fontWeight = FontWeight.Bold
                )
            }
        }
    }
}
